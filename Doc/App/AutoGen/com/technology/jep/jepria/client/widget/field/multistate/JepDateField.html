    public class JepDateField
extends JepMultiStateField<MaskedDateBox,com.google.gwt.user.client.ui.HTML>
    Поле для ввода даты.
    Использует маску ввода.
    Для задания нестандартного (отличного от dd.MM.yyyy) формата ввода и вывода даты необходимо в классе-наследнике переопределить метод addEditableCard() следующим образом (задаётся формат dd.MM.yyyy HH:mm:ss): 
       JepDateField dateField = new JepDateField("Date field"){
     @Override
     protected void addEditableCard() {
       format = DateTimeFormat.getFormat(DEFAULT_DATE_FORMAT + " " + DEFAULT_TIME_FORMAT);
       Mask mask = new Mask(DEFAULT_DATE_FORMAT_MASK + " " + DEFAULT_TIME_FORMAT_MASK);
       editableCard = new MaskedDateBox(new DatePicker(), null, new XDefaultFormat(format), mask);
       editablePanel.add(editableCard);
     }};
