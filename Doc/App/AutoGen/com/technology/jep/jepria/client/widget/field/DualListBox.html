    public class DualListBox
extends com.google.gwt.user.client.ui.Composite
implements com.google.gwt.user.client.ui.HasWidgets, com.google.gwt.event.logical.shared.HasValueChangeHandlers<java.util.List<JepOption>>
    Компонент редактирования для поля JepDualListField. Представляет собой два списка. Левый список содержит список опций, правый - список выбранных опций. Элементы перемещаются между списками либо с помощью расположенного между списками блока кнопок, либо двойным кликом.
